# Printing-Pattern
/*Enter the number of row and column: 4
**** 
**** 
**** 
**** */


#include<iostream>
using namespace std;
int main()
{
    int a;
    cout<<"Enter the number of row and column: ";
    cin>>a;
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<a;j++)
        {
            cout<<"*";
        }

    cout<<endl;

    }
    }
